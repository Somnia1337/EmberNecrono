class Solution {
    public int minGroups(int[][] intervals) {
        Arrays.sort(intervals, Comparator.comparingInt(a -> a[0]));
        Queue<Integer> pq = new PriorityQueue<>();
        for (int[] itv : intervals) {
            if (!pq.isEmpty() && pq.peek() < itv[0]) pq.poll();
            pq.offer(itv[1]);
        }
        return pq.size();
    }
