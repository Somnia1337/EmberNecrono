#include <stdio.h>
#include <math.h>

int N[100][100];

int main()
{
    int n, m, k, x, y, a, b, i, j, s, t, c = 0;
    scanf("%d %d %d\n", &n, &m, &k);
    for (i = 0; i < n; i++)
    {
        for (j = 0; j < n; j++)
        {
            N[i][j] = 1;
        }
    }
    for (i = 0; i < m; i++)
    {
        scanf("%d %d\n", &x, &y);
        N[x - 1][y - 1] = -1;
    }
    for (i = 0; i < k; i++)
    {
        scanf("%d %d\n", &x, &y);
        N[x - 1][y - 1] = -2;
    }
    for (i = 0; i < n; i++)
    {
        for (j = 0; j < n; j++)
        {
            if (N[i][j] == -1)
            {
                x = i, y = j;
                for (s = 0; s < n; s++)
                {
                    for (t = 0; t < n; t++)
                    {
                        if (abs(s - x) + abs(t - y) >= 1 && abs(s - x) + abs(t - y) <= 2)
                        {
                            if (N[s][t] == 1)
                            {
                                N[s][t] = 0;
                            }
                        }
                    }
                }
            }
            else if (N[i][j] == -2)
            {
                x = i, y = j;
                for (s = 0; s < n; s++)
                {
                    for (t = 0; t < n; t++)
                    {
                        if (abs(s - x) + abs(t - y) >= 1 && abs(s - x) + abs(t - y) <= 2)
                        {
                            if (N[s][t] == 1)
                            {
                                N[s][t] = 0;
                            }
                        }
                        else if (abs(s - x) + abs(t - y) == 3 && x != s && y != t)
                        {
                            if (N[s][t] == 1)
                            {
                                N[s][t] = 0;
                            }
                        }
                        else if (abs(s - x) == 2 && abs(t - y) == 2)
                        {
                            if (N[s][t] == 1)
                            {
                                N[s][t] = 0;
                            }
                        }
                    }
                }
            }
        }
    }
    for (i = 0; i < n; i++)
    {
        for (j = 0; j < n; j++)
        {
            if (N[i][j] == 1)
            {
                c++;
            }
        }
    }
    printf("%d", c);

    return 0;
}
