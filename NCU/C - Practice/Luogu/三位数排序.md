[【深基3.例8】三位数排序 - 洛谷](https://www.luogu.com.cn/problem/P5715)

```c
#include <stdio.h>

int main()
{
	int a, b, c, A, B, C;
	scanf("%d %d %d", &a, &b, &c);
	if (a == b && b == c)
	{
		printf("%d %d %d",a,b,c);
		return 0;
	}
	if (b <= a && b <= c)
	{
		A = b;
	}
	if (c <= a && c <= b)
	{
		A = c;
	}
	if (a <= b && a <= c)
	{
		A = a;
	}
	if (b > a && b > c)
	{
		C = b;
	}
	if (c > a && c > b)
	{
		C = c;
	}
	if (a > b && a > c)
	{
		C = a;
	}
	if (a >= b && a <= c || a >= c && a <= b)
	{
		B = a;
	}
	if (b >= a && b <= c || b <= a && b >= c)
	{
		B = b;
	}
	if (c >= a && c <= b || c <= a && c >= b)
	{
		B = c;
	}
	printf("%d %d %d", A, B, C);
	return 0;
}
```
