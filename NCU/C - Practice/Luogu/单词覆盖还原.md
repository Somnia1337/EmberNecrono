[单词覆盖还原 - 洛谷](https://www.luogu.com.cn/problem/P1321)

```c
//解 after 题解

#include <stdio.h>

int main()
{
	char L[280];
	int i, cnt, b = 0, g = 0;
	char n = '0';
	for (i = 0; n != EOF; i++)
	{
		L[i] = n;
		n = getchar();
	}
	cnt = i;
	for (i = 0; i < cnt; i++)
	{
		if (L[i] == 'b' || L[i + 1] == 'o' || L[i + 2] == 'y')
		{
			b++;
		}
		if (L[i] == 'g' || L[i + 1] == 'i' || L[i + 2] == 'r' || L[i + 3] == 'l')
		{
			g++;
		}
	}
	printf("%d\n%d", b, g);

	return 0;
}
```

运用两个if里面的或门，不管单词被覆盖了多少，都可以正确识别每一个单词
