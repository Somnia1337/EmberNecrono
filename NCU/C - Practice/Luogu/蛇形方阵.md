[【深基5.习6】蛇形方阵 - 洛谷](https://www.luogu.com.cn/problem/P5731)

```c
#include <stdio.h>

int main()
{
	int A[9][9];
	int n, i = 0, j, x = -1, y = 0, c = 1, b;
	scanf("%d", &n);
	while (1)
	{
		if (i == n * n)
		{
			break;
		}
		b = n - c / 2;
		if (c % 4 == 1)
		{
			x++;
			for (j = x; j < x + b; j++)
			{
				i++;
				A[y][j] = i;
			}
			x = x + b - 1;
			c++;
			continue;
		}
		else if (c % 4 == 2)
		{
			y++;
			for (j = y; j < y + b; j++)
			{
				i++;
				A[j][x] = i;
			}
			y = y + b - 1;
			c++;
			continue;
		}
		else if (c % 4 == 3)
		{
			x--;
			for (j = x; j > x - b; j--)
			{
				i++;
				A[y][j] = i;
			}
			x = x - b + 1;
			c++;
			continue;
		}
		else if (c % 4 == 0)
		{
			y--;
			for (j = y; j > y - b; j--)
			{
				i++;
				A[j][x] = i;
			}
			y = y - b + 1;
			c++;
			continue;
		}
	}
	for (i = 0; i < n; i++)
	{
		for (j = 0; j < n; j++)
		{
			printf("%3d", A[i][j]);
		}
		printf("\n");
	}

	return 0;
}
```
