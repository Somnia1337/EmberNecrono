```c
//解1

#include <stdio.h>

int main()
{
    int s, v, y, h, m, H, M;
    scanf("%d %d", &s, &v);
    m = s / v;
    y = s % v;
    if (y != 0)
    {
        m++;
    }
    m += 10;
    h = m / 60;
    m = m - 60 * h;
    M = 60 - m;
    H = 7 - h;
    if (H < 0)
    {
        H += 24;
    }
    printf("%.2d:%.2d", H, M);

    return 0;
}
```
