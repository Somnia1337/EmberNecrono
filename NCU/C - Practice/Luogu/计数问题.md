[[NOIP2013 普及组] 计数问题 - 洛谷](https://www.luogu.com.cn/problem/P1980)

```c
//解1

#include <stdio.h>
#include <math.h>

int lengthOf(int a)
{
    int c;
    for (c = 0; a >= 1; c++)
    {
        a = a / 10;
    }
    return c;
}
int main()
{
    long long x, n, cnt = 0, m = 1, N[7];
    scanf("%lld %lld", &n, &x);
    int k;
    
    for (; m <= n;)
    {
        int len = lengthOf(m);
        long long M = pow(10, len);
        for (; m < M && m <= n; m++)
        {
            for (int i = 0, k = 1; i < len; i++, k++)
            {
                long long j = pow(10, k);
                if (k < 2)
                {
                    N[i] = m % j;
                }
                else
                {
                    N[i] = m / (j / 10);
                }
                while (N[i] >= 10)
                {
                    N[i] = N[i] % 10;
                }
            }
            for (int i = 0; i < len; i++)
            {
                if (N[i] == x)
                {
                    cnt++;
                }
            }
        }
    }

    printf("%lld", cnt);
    return 0;
}
```

```c
#include <stdio.h>

int main(void)
{
	int n, x, i, a, t;
	scanf("%d %d", &n, &x);
	t = 0;
	for (i = 1; i <= n; i++)
	{
		a = i;
		for (; a > 0; )
		{
			if (a % 10 == x)
			{
				t++;
			}
			a = a / 10;
		}
	}
	printf("%d", t);
}
```
