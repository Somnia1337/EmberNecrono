[语句解析 - 洛谷](https://www.luogu.com.cn/problem/P1597)

```c
//解 after 题解

#include <stdio.h>

int main()
{
	int N[3] = { 0,0,0 };
	char m, n;
	while (1)
	{
		m = getchar();
		if (m >= 97 && m <= 99 || m >= 48 && m <= 57)
		{
			getchar();
			getchar();
			n = getchar();
			N[m - 'a'] = (n >= '0' && n <= '9') ? n - '0' : N[n - 'a'];
			getchar();
		}
		else
		{
			break;
		}
	}
	printf("%d %d %d", N[0], N[1], N[2]);

	return 0;
}
```
