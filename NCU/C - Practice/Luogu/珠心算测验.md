[[NOIP2014 普及组] 珠心算测验 - 洛谷](https://www.luogu.com.cn/problem/P2141)

```c
#include <stdio.h>

int main()
{
	int D[100], M[100];
	int n, i, j, k, a, c = 0, m, l, b;
	scanf("%d", &n);
	for (i = 0; i < 100; i++)
	{
		M[i] = 0;
	}
	for (i = 0; i < n; i++)
	{
		scanf("%d", &D[i]);
		getchar();
	}
	m = 0;
	for (i = 0; i < n; i++)
	{
		for (j = i + 1; j < n; j++)
		{
			a = D[i] + D[j];
			for (k = 0; k < n; k++)
			{
				if (a == D[k])
				{
					b = 1;
					for (l = 0; l < m + 1; l++)
					{
						if (a == M[l])
						{
							b = 0;
							break;
						}
					}
					if (b == 1)
					{
						c++;
						M[m] = a;
						m++;
					}
				}
			}
		}
	}
	printf("%d", c);

	return 0;
}
```
