```c
//解1

#include <stdio.h>

int main()
{
	int x, n, v, w, r;
	scanf("%d %d", &x, &n);
	w = n / 7;
	r = n % 7;
	if ((x + r) < 7)
	{
		v = w * 5 + r;
	}
	else if ((x + r) == 7)
	{
		if (x == 7)
		{
			v = w * 5 + r;
		}
		else
		{
			v = w * 5 + r - 1;
		}
	}
	else if ((x + r) >= 8)
	{
		if (x == 7)
		{
			v = w * 5 + r - 1;
		}
		else
		{
			v = w * 5 + r - 2;
		}
	}
	printf("%d", v * 250);
	return 0;
}
```
