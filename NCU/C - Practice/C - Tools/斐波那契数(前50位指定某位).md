#include <stdio.h>

int main()
{
    int n, i;
    scanf("%d", &n);
    long long str[52];
    str[0] = 1, str[1] = 1;
    for (i = 1; i <= 51; i++)
    {
        str[i + 1] = str[i] + str[i - 1];
        if (i == n)
        {
            break;
        }
    }
    printf("%lld", str[n - 1]);

    return 0;
}
