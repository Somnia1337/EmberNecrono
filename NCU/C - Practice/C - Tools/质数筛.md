//打印形式

#include <stdio.h>
#include <math.h>

int main()
{
	int n, i, j, v;
	scanf("%d", &n);
	for (i = 1; i <= n; i += 2)
	{
		if (i < 2)
		{
			printf("2\n");
			continue;
		}
		v = 1;
		for (j = 2; j <= sqrt(i); j++)
		{
			if (i % j == 0)
			{
				v = 0;
				break;
			}
		}
		if (v == 1)
		{
			printf("%d\n", i);
		}
	}

	return 0;
}

//库形式

#include <stdio.h>
#include <math.h>

int main()
{
	int Z[10000];
	int n, i, j, v, k = 0;
	scanf("%d", &n);
	for (i = 1; i <= n; i+=2)
	{
		if (i < 2)
		{
			Z[k] = 2;
			k++;
			continue;
		}
		v = 1;
		for (j = 2; j <= sqrt(i); j++)
		{
			if (i % j == 0)
			{
				v = 0;
				break;
			}
		}
		if (v == 1)
		{
			Z[k] = i;
			k++;
		}
	}

	return 0;
}
