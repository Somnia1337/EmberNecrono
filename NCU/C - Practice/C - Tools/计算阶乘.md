```c
//循环法

#include <stdio.h>
#include <stdlib.h>
#include <corecrt_math.h>

int main()
{
	int n;
	printf("输入个数：");
	scanf("%d",&n);
	int x;
	int s = 1;
	for (x = 1; x <= n;)
	{
		s = s * x;
		x++;
	}
	printf("%d\n", s);
  
	return 0;
}
```

```c
//函数递归法

#include <stdio.h>

int f(int n)
{
  if (n == 0 || n == 1)
  {
    return 1;
  }
  return n * f(n - 1);
}

int main()
{
  int result,x;
  scanf("%d",&x);
  result = f(x);
  printf("%d\n",result);
  
  return 0;
}
```
