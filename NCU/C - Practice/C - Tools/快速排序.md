#include <stdio.h>

int N[1000001];

void rapidsort(int l, int r)
{
    int mid = N[(l + r) / 2];
    int i = l, j = r;
    do {
        while (N[i] < mid) i++;
        while (N[j] > mid) j--;
        if (i <= j)
        {
            int t = N[i];
            N[i] = N[j];
            N[j] = t;
            i++;
            j--;
        }
    } while (i <= j);
    if (l < j) rapidsort(l, j);
    if (i < r) rapidsort(i, r);
}

int main()
{
    int n, max;
    int i, j;
    scanf("%d", &n);
    for (i = 0; i < n; i++)
    {
        scanf("%d", &N[i]);
    }
    max = N[0];
    for (i = 0; i < n; i++)
    {
        if (N[i] > max)
        {
            max = N[i];
        }
    }
    rapidsort(0, n);
    for (i = 1; i < n; i++)
        N[i - 1] = N[i];
    N[n - 1] = max;
    for (i = 0; i < n; i++)
        printf("%d ", N[i]);

    return 0;
}

#include <stdio.h>

int N[100];

void swap(int* x, int* y)
{
	int t = *x;
	*x = *y;
	*y = t;
}

void rapidSort(int lower, int upper)
{
	
	int k = N[lower], c = lower + 1;
	for (int i = lower + 1; i <= upper; i++)
		if (N[i] <= k)
			swap(&N[i], &N[c]), c++;
	c--;
	swap(&N[lower], &N[c]);
	if (lower < c - 1) rapidSort(lower, c - 1);
	if (c + 1 < upper) rapidSort(c + 1, upper);
}

int main()
{
	int n;
	int i;
	scanf("%d", &n);
	for (i = 0; i < n; i++)
		scanf("%d", &N[i]);
	rapidSort(0, n - 1);
	for (i = 0; i < n; i++)
		printf("%d ", N[i]);

	return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <time.h>

int N[100000];

void swap(int* x, int* y)
{
    int t = *x;
    *x = *y;
    *y = t;
}

void quickSort(int a[], int left, int right)
{
    if (left >= right)
        return;
    int i = left, j = right, pivot = rand() % (right - left + 1) + left;
        swap(&a[left], &a[pivot]);
        while (i < j)
        {
            while (j > i && a[j] >= a[left])
                j--;
            while (i < j && a[i] <= a[left])
                i++;
            swap(&a[i], (i == j) ? &a[left] : &a[j]);
        }
    quickSort(a, left, i - 1);
    quickSort(a, j + 1, right);
}

int main()
{
    srand(time(NULL));
	int n;
	int i;
	scanf("%d", &n);
	for (i = 0; i < n; i++)
		scanf("%d", &N[i]);
	quickSort(N, 0, n - 1);
	for (i = 0; i < n; i++)
		printf("%d ", N[i]);

	return 0;
}
