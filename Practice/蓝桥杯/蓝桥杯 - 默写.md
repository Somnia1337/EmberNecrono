private static long pow(long x, long p) {
	long ret = 1;
	while (p > 0) {
		if ((p & 1) == 1) {
			ret = ((ret % MOD) * (x % MOD)) % MOD;
		}
		x = ((x % MOD) * (x % MOD)) % MOD;
		p >>= 1;
	}
	return ret;
}
